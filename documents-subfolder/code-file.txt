      # 🧭 Навігація
      cd Ім’яПапки        # зайти в папку
      cd ..               # на рівень вгору
      dir                 # показати вміст поточної папки
      
      # 📄 Файли
      echo "Привіт!" > файл.txt    # створити файл із текстом
      type файл.txt                # показати вміст файлу
      del файл.txt                 # видалити файл
      
      # 📁 Папки
      mkdir НоваПапка             # створити папку
      rmdir НазваПапки            # видалити порожню папку
      
      # 🛠 Інше
      cls                         # очистити термінал
      exit                        # закрити PowerShell
      
      # 📜 Bash-стиль команди
      ls                          # показати вміст (аналог dir)
      ls -Force                   # показати всі файли, включно з прихованими
      pwd                         # поточна директорія
      cat файл.txt                # вивести вміст файлу
      
      # 🧩 Git-команди
      git config --global user.name "Твоє Ім’я"
      git config --global user.email "email@example.com"
      git status                  # статус Git-репозиторію
      git add файл.txt            # додати файл до індексу
      git add .                   # додати всі зміни
      git commit -m "опис"        # зробити коміт з повідомленням
      git log                     # історія комітів
      git checkout branch-name    # перейти на гілку
      git checkout -b new-branch  # створити і перейти на нову гілку
      git checkout файл.txt       # скасувати зміни у файлі до останнього коміту
